def strip_accents(value):
    # Décompose les caractères accentués (é → e + accent) puis supprime les marques diacritiques
    return "".join(
        ch for ch in unicodedata.normalize("NFKD", str(value))
        if not unicodedata.combining(ch)
    )


def clean_string(value):
    # Retourne une chaîne vide si la valeur est NaN, sinon supprime les accents, met en minuscules et compresse les espaces
    if pd.isna(value):
        return ""
    return re.sub(r"\s+", " ", strip_accents(str(value)).strip().lower())


def is_bad_country(value):
    # Retourne True si la valeur de pays est dans la liste des valeurs invalides
    return clean_string(value) in BAD_COUNTRY_VALUES


def normalize_country(value):
    # Normalise un nom de pays : retourne NaN si invalide, cherche dans les alias, sinon met en title case
    key = clean_string(value)
    if not key or key in BAD_COUNTRY_VALUES:
        return np.nan
    if key in COUNTRY_ALIASES:
        return COUNTRY_ALIASES[key]
    if len(key) == 2:                # Un code à 2 lettres non reconnu est ignoré (trop ambigu)
        return np.nan
    return str(value).strip().title()


def infer_country_from_source(source_code):
    # Tente de déduire le pays à partir du code source en cherchant dans SOURCE_COUNTRY_MAP et COUNTRY_ALIASES
    source = clean_string(source_code)
    if not source:
        return np.nan
    if source in SOURCE_COUNTRY_MAP:      # Correspondance exacte sur le code source complet
        return SOURCE_COUNTRY_MAP[source]
    for part in re.split(r"[_\-\s\.]+", source):    # Découpe le code source en segments séparés par _, -, espace ou .
        if part in SOURCE_COUNTRY_MAP:
            return SOURCE_COUNTRY_MAP[part]
        if part in COUNTRY_ALIASES:
            return COUNTRY_ALIASES[part]
    for pattern, country in SOURCE_COUNTRY_MAP.items():    # Cherche si un pattern connu est contenu dans le code source
        if clean_string(pattern) and clean_string(pattern) in source:
            return country
    return np.nan


def infer_country_from_language(lang):
    """[P4] Last resort: map language code to country."""
    # Dernier recours : convertit le code langue en pays via LANGUAGE_TO_COUNTRY
    key = clean_string(lang)
    if not key or key in ("", "unknown", "nan"):
        return np.nan
    prefix = key[:2]      # Prend les 2 premiers caractères du code langue (ex: "fr-FR" → "fr")
    return LANGUAGE_TO_COUNTRY.get(prefix, LANGUAGE_TO_COUNTRY.get(key, np.nan))


def standardize_country(row):
    """Priority: country field → source code → language."""
    # Résolution du pays en 3 étapes : champ country → code source → langue
    country = normalize_country(row.get("country", np.nan))
    if pd.notna(country):
        return country
    from_source = infer_country_from_source(row.get("source_code", ""))
    if pd.notna(from_source):
        return from_source
    # [P4] En dernier recours, on utilise la langue
    return infer_country_from_language(row.get("language", ""))


def find_numeric_col(df, candidates):
    """Return first column in df that matches any candidate name and has non-zero data."""
    # Cherche la première colonne numérique du DataFrame correspondant à un des noms candidats
    lower_map = {c.lower(): c for c in df.columns}   # Crée un dictionnaire nom_minuscule → nom_réel des colonnes
    for name in candidates:
        if name.lower() in lower_map:
            col = lower_map[name.lower()]
            converted = pd.to_numeric(df[col], errors="coerce")    # Convertit la colonne en numérique, les erreurs deviennent NaN
            if converted.notna().sum() > 0 and converted.fillna(0).sum() > 0:   # Vérifie que la colonne a des données non nulles
                return col
    # Recherche approximative : cherche si un des tokens candidats est contenu dans le nom d'une colonne
    for col in df.columns:
        c = col.lower()
        if any(token in c for token in candidates):
            converted = pd.to_numeric(df[col], errors="coerce")
            if converted.notna().sum() > 0 and converted.fillna(0).sum() > 0:
                return col
    return None


def find_col(df, candidates):
    # Cherche la première colonne du DataFrame correspondant à un des noms candidats (sans vérifier si elle est numérique)
    lower_map = {c.lower(): c for c in df.columns}
    for name in candidates:
        if name.lower() in lower_map:
            return lower_map[name.lower()]
    for col in df.columns:
        if any(token in col.lower() for token in candidates):
            return col
    return None


def to_bool(series):
    # Convertit une colonne en booléen : gère les types bool natifs et les chaînes "true/1/yes/oui/vrai"
    if series.dtype == bool:
        return series.fillna(False)
    return series.astype(str).str.strip().str.lower().isin(
        ["true", "1", "yes", "y", "oui", "vrai"]
    )


def normalize_0_100(series):
    # Normalise une série numérique sur une échelle de 0 à 100 (min-max scaling)
    s = pd.to_numeric(series, errors="coerce").fillna(0)
    mn, mx = s.min(), s.max()
    if mx == mn:                          # Si toutes les valeurs sont identiques, on retourne 50 pour tous
        return pd.Series([50.0] * len(s), index=s.index)
    return ((s - mn) / (mx - mn) * 100).round(2)


def safe_pct(numerator, denominator):
    # Calcule un pourcentage en évitant la division par zéro ou par NaN
    if not denominator or pd.isna(denominator):
        return np.nan
    return round(100 * numerator / denominator, 2)


def count_keywords(texts, keyword_dict):
    # Compte combien de textes contiennent chaque terme (max 1 comptage par texte même si le terme apparaît plusieurs fois)
    counts = {term: 0 for term in keyword_dict}
    for text in texts:
        t = clean_string(text)
        for term, patterns in keyword_dict.items():
            for pat in patterns:
                if re.search(pat, t, flags=re.IGNORECASE):   # Dès qu'un pattern correspond, on compte +1 et on passe au terme suivant
                    counts[term] += 1
                    break
    return counts


# ---------------------------------------------------------------------------
# [P1] Aggressive date parser — tries 8 formats + Unix timestamps
# ---------------------------------------------------------------------------

# Liste des formats de date à tester dans l'ordre lors du parsing agressif
DATE_FORMATS = [
    "%Y-%m-%d %H:%M:%S",
    "%Y-%m-%dT%H:%M:%S",
    "%Y-%m-%d",
    "%d/%m/%Y %H:%M:%S",
    "%d/%m/%Y",
    "%m/%d/%Y",
    "%d-%m-%Y",
    "%d.%m.%Y",
    "%B %d, %Y",
    "%d %B %Y",
    "%b %d, %Y",
    "%Y%m%d",
]


def parse_dates_aggressive(series):
    # Étape 1 : pandas tente de détecter automatiquement le format (couvre ISO 8601 et la plupart des formats standards)
    parsed = pd.to_datetime(series, errors="coerce", infer_datetime_format=True)
    n_ok = parsed.notna().sum()    # Compte le nombre de dates correctement parsées

    # Étape 2 : si moins de 50% des dates sont parsées, on réessaie avec dayfirst=True pour les formats européens DD/MM
    if n_ok < len(series) * 0.5:
        parsed2 = pd.to_datetime(series, errors="coerce", dayfirst=True)
        if parsed2.notna().sum() > n_ok:    # On garde le résultat dayfirst seulement s'il parse plus de dates
            parsed = parsed2
            n_ok = parsed.notna().sum()

    # Étape 3 : pour les dates encore NaT, on teste chaque format explicite de la liste DATE_FORMATS
    for fmt in DATE_FORMATS:
        mask = parsed.isna()      # Identifie les lignes dont la date n'est pas encore parsée
        if mask.sum() == 0:
            break
        try:
            fixed = pd.to_datetime(series[mask], format=fmt, errors="coerce")
            parsed[mask] = fixed   # Remplace les NaT restants par les dates parsées avec ce format
        except Exception:
            pass

    # Étape 4 : pour les dates encore NaT, on tente de les interpréter comme des timestamps Unix (secondes depuis 1970)
    mask = parsed.isna()
    if mask.any():
        numeric = pd.to_numeric(series[mask], errors="coerce")
        valid_unix = numeric[(numeric > 1e8) & (numeric < 2e10)]   # Filtre les valeurs dans une plage Unix réaliste
        if not valid_unix.empty:
            from_unix = pd.to_datetime(valid_unix, unit="s", errors="coerce")
            parsed[from_unix.index] = from_unix

    return parsed

        comments  =("message_type",   lambda x: (x == "comment").sum()),           # Nombre de commentaires
    )
    .reset_index()
    .rename(columns={"country_clean": "country"})
)


# ---------------------------------------------------------------------------
# Sentiment cross-tabs for Excel
# ---------------------------------------------------------------------------

# Calcule la distribution des sentiments par pays pour l'onglet Excel "Sentiment by Country"
sentiment_by_country = (
    country_df.groupby(["country_clean", "sentiment_label"])
    .size()
    .reset_index(name="count")
    .rename(columns={"country_clean": "country"})
)
if not sentiment_by_country.empty:
    totals = sentiment_by_country.groupby("country")["count"].sum().reset_index(name="total")
    sentiment_by_country = sentiment_by_country.merge(totals, on="country")
    sentiment_by_country["percentage"] = (
        100 * sentiment_by_country["count"] / sentiment_by_country["total"]
    ).round(2)   # Calcule le pourcentage de chaque sentiment au sein de chaque pays

# Regroupe les langues en 3 catégories : FR, EN, et Other pour simplifier l'analyse
df["lang_group"] = np.where(
    df["language"] == "fr", "FR",
    np.where(df["language"] == "en", "EN", "Other"),
)
# Calcule la distribution des sentiments par groupe de langue pour l'onglet "Sentiment by Language"
sentiment_by_lang = df.groupby(["lang_group", "sentiment_label"]).size().reset_index(name="count")
lang_totals = sentiment_by_lang.groupby("lang_group")["count"].sum().reset_index(name="total")
sentiment_by_lang = sentiment_by_lang.merge(lang_totals, on="lang_group")
sentiment_by_lang["percentage"] = (
    100 * sentiment_by_lang["count"] / sentiment_by_lang["total"]
).round(2)   # Calcule le pourcentage de chaque sentiment au sein de chaque groupe de langue
