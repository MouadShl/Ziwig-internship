    should_stop = True

def load_config(path: str) -> dict:
    with open(path, "r", encoding="utf-8") as f:
        return json.load(f)


def append_jsonl(path: Path, obj: dict):
    with open(path, "a", encoding="utf-8") as f:
        f.write(json.dumps(obj, ensure_ascii=False) + "\n")


def build_output_files(cfg: dict):
    output_dir = Path(cfg["output"]["dir"])
    output_dir.mkdir(parents=True, exist_ok=True)
    stamp = datetime.now().strftime("%d%m%Y_%Hh%M")
    source_id = cfg["source_id"]
    posts_file = output_dir / f"{source_id}_post_and_comment_{stamp}.jsonl"
    errors_file = output_dir / f"{source_id}_errors_{stamp}.jsonl"
    return posts_file, errors_file


def clean_text(value: str) -> str:
    value = value or ""
    value = value.replace("\xa0", " ")
    value = re.sub(r"\s+", " ", value)
    return value.strip()


def clean_multiline_text(value: str) -> str:
    value = value or ""
    value = value.replace("\xa0", " ")
    lines = [re.sub(r"\s+", " ", x).strip() for x in value.splitlines()]
    lines = [x for x in lines if x]
    return "\n".join(lines).strip()


def safe_int(value, default=0):
    if value is None:
        return default
    s = re.sub(r"[^\d]", "", str(value))
    return int(s) if s.isdigit() else default


def fetch_html(session: requests.Session, url: str, timeout: int) -> str:
    r = session.get(url, timeout=(10, timeout))
    r.raise_for_status()
    return r.text

# DOCTISSIMO HELPERS
# =========================================================

DOCTISSIMO_THREAD_RE = re.compile(r"sujet_(\d+)", re.I)
DOCTISSIMO_THREAD_PAGE_RE = re.compile(r"(.*?_)(\d+)(\.htm)$", re.I)

BAD_BODY_EXACT = {
    "Alerter Copier le lien Le lien a été copié dans votre presse-papier",
    "Alerter",
    "Copier le lien",
    "Le lien a été copié dans votre presse-papier",
}

BAD_BODY_CONTAINS = [
    "Alerter Copier le lien",
    "Le lien a été copié dans votre presse-papier",
]

BAD_AUTHOR_VALUES = {
    "",
    "Alerter",
    "Copier le lien",
}

DATE_VISIBLE_RE = re.compile(r"\d{2}/\d{2}/\d{4}\s+à\s+\d{2}h\d{2}")


def build_doctissimo_listing_url(seed_url: str, page_num: int) -> str:
    if page_num == 1:
        return seed_url

    # liste_sujet-1.htm -> liste_sujet-2.htm
    m = re.search(r"(.*liste_sujet-)(\d+)(\.htm)$", seed_url, re.I)
    if m:
        return f"{m.group(1)}{page_num}{m.group(3)}"

    return seed_url


def extract_thread_id_from_url(url: str) -> str:
    m = DOCTISSIMO_THREAD_RE.search(url or "")
    return m.group(1) if m else ""


def build_doctissimo_thread_page_url(first_page_url: str, page_num: int) -> str:
    """
    Example:
    https://forum.doctissimo.fr/sante/Endometriose/endometriose-sujet_7362_1.htm
    -> page 2
    https://forum.doctissimo.fr/sante/Endometriose/endometriose-sujet_7362_2.htm
    """
    if page_num == 1:
        return first_page_url

    m = DOCTISSIMO_THREAD_PAGE_RE.search(first_page_url)
    if m:
        return f"{m.group(1)}{page_num}{m.group(3)}"

    return first_page_url


def extract_visible_date_from_block(block) -> str:
    txt = block.get_text(" ", strip=True)
    m = DATE_VISIBLE_RE.search(txt)
    return m.group(0) if m else ""


def extract_iso_date_from_block(block) -> str:
    meta = block.find("meta", attrs={"itemprop": "dateModified"})
    if meta and meta.get("content"):
        return clean_text(meta.get("content"))
    return ""


def extract_author_and_user_id(block):
    author = ""
    native_user_id = ""

    # Best source: button with data-id-user
    user_btn = block.find("button", attrs={"data-id-user": True})
    if user_btn:
        native_user_id = clean_text(user_btn.get("data-id-user", ""))

        # visible username may be inside the button
        btn_text = clean_text(user_btn.get_text(" ", strip=True))
        if btn_text and btn_text not in BAD_AUTHOR_VALUES:
            author = btn_text

        # sometimes data-params contains id_user
        if not native_user_id:
            data_params = user_btn.get("data-params", "")
            m = re.search(r'"id_user":"?(\d+)"?', data_params)
            if m:
                native_user_id = m.group(1)

    # fallback: profile links
    if not author:
        for a in block.find_all("a", href=True):
            href = a.get("href", "")
            txt = clean_text(a.get_text(" ", strip=True))
            if not txt:
                continue
            if "profil" in href.lower() or "membre" in href.lower():
                author = txt
                break

    # fallback: common username places
    if not author:
        for sel in [
            ".md-name",
            ".user-login",
            ".username",
            ".pseudo",
            ".author",
            "[itemprop='author']",
        ]:
            el = block.select_one(sel)
            if el:
                txt = clean_text(el.get_text(" ", strip=True))
                if txt and txt not in BAD_AUTHOR_VALUES:
                    author = txt
                    break

    author = clean_text(author)
    user_id = native_user_id if native_user_id else author

    return author, user_id, native_user_id


def remove_unwanted_nodes(block):
    # Remove obvious UI / actions / toolbar / signatures / quotes that pollute the body
    selectors_to_remove = [
        ".md-tools",
        ".md-toolbar",
        ".toolbar",
        ".post-toolbar",
        ".actions",
        ".action",
        ".user-actions",
        ".btn",
        ".picto-msg-quote",
        ".signature",
        ".avatar",
        ".profile-card",
        ".quote-button",
        ".share",
        ".alert",
        ".report",
        ".spoiler-title",
    ]

    for selector in selectors_to_remove:
        for el in block.select(selector):
            el.decompose()

    # remove scripts/styles
    for el in block.find_all(["script", "style", "noscript"]):
        el.decompose()


def extract_body_from_block(block) -> str:
    """
    IMPORTANT:
    We must extract only the real message content, not the full block text.
    """

    # Work on a copy-like subtree by mutating the block directly after we already extracted IDs
    remove_unwanted_nodes(block)

    candidate_selectors = [
        ".post_message",
        ".message",
        ".content",
        ".md-message",
        ".md-post-content",
        ".post-content",
        ".txt-msg",
        ".entry-content",
        ".forum-message",
    ]

    # 1) Best case: real content node exists
    for selector in candidate_selectors:
        el = block.select_one(selector)
        if el:
            text = clean_multiline_text(el.get_text("\n", strip=True))
            if text:
                return text

    # 2) Fallback: collect paragraph-like text, skip short UI garbage
    parts = []
    for el in block.find_all(["p", "div", "span", "blockquote", "li"]):
        txt = clean_multiline_text(el.get_text("\n", strip=True))
        if not txt:
            continue
        if txt in BAD_BODY_EXACT:
            continue
        if any(bad in txt for bad in BAD_BODY_CONTAINS):
            continue
        if len(txt) < 3:
            continue
        parts.append(txt)

    # Keep unique parts in order
    seen = set()
    ordered_parts = []
    for p in parts:
        if p not in seen:
            seen.add(p)
            ordered_parts.append(p)

    body = "\n".join(ordered_parts).strip()

    # Final clean
    body = re.sub(r"\n{3,}", "\n\n", body).strip()

    return body


def is_bad_body(body: str) -> bool:
    if not body:
        return True

    compact = clean_text(body)

    if compact in BAD_BODY_EXACT:
        return True

    if any(bad in compact for bad in BAD_BODY_CONTAINS):
        return True

    # body that is only toolbar noise
    if len(compact) < 8:
        return True

    return False


def extract_opening_thread_title(soup: BeautifulSoup) -> str:
    for sel in ["h1", ".topic-title", ".subject-title", "title"]:
        el = soup.select_one(sel)
        if el:
            txt = clean_text(el.get_text(" ", strip=True))
            if txt:
                return txt
    return ""


def extract_thread_pages_count(soup: BeautifulSoup, first_page_url: str) -> int:
    max_page = 1

    # Find links that point to sujet_<threadid>_<page>.htm
    thread_id = extract_thread_id_from_url(first_page_url)
    if not thread_id:
        return 1

    for a in soup.find_all("a", href=True):
        href = a.get("href", "")
        if f"sujet_{thread_id}_" not in href:
            continue
        m = re.search(rf"sujet_{thread_id}_(\d+)\.htm", href)
        if m:
            max_page = max(max_page, int(m.group(1)))

    return max_page


def extract_threads_from_listing(soup: BeautifulSoup, listing_url: str):
    threads = []
    seen_thread_ids = set()

    for a in soup.find_all("a", href=True):
        href = a.get("href", "")
        if "sujet_" not in href:
            continue

        full_url = urljoin(listing_url, href)
        thread_id = extract_thread_id_from_url(full_url)
        if not thread_id:
            continue

        title = clean_text(a.get_text(" ", strip=True))
        if not title:
            continue

        if thread_id in seen_thread_ids:
            continue
        seen_thread_ids.add(thread_id)

        threads.append({
            "thread_id": thread_id,
            "thread_url_id": thread_id,
            "thread_title": title,
            "thread_url": full_url,
            "last_page": 1,
            "listing_author": "",
            "listing_author_id": "",
            "replies_count": 0,
            "views_count": 0,
            "last_message_date": "",
            "last_message_author": "",
            "last_message_author_id": "",
        })

    return threads


def parse_doctissimo_thread(session, thread_url: str, timeout: int, sleep_seconds: float):
    """
    Parse all pages of one thread and deduplicate posts by native data-id_post.
    """
    first_html = fetch_html(session, thread_url, timeout)
    first_soup = BeautifulSoup(first_html, "lxml")

    thread_id = extract_thread_id_from_url(thread_url)
    thread_title_detail = extract_opening_thread_title(first_soup)
    pages_count = extract_thread_pages_count(first_soup, thread_url)

    all_posts = []
    seen_native_post_ids = set()

    for page_num in range(1, pages_count + 1):
        page_url = build_doctissimo_thread_page_url(thread_url, page_num)
        if page_num == 1:
            soup = first_soup
        else:
            html = fetch_html(session, page_url, timeout)
            soup = BeautifulSoup(html, "lxml")
            time.sleep(sleep_seconds)

        post_blocks = soup.select("div.md-post[data-id_post]")

        for block in post_blocks:
            native_post_id = clean_text(block.get("data-id_post", ""))
            if not native_post_id:
                continue

            # CRITICAL DEDUPE
            if native_post_id in seen_native_post_ids:
                continue
            seen_native_post_ids.add(native_post_id)

            anchor_el = block.find("span", id=True)
            anchor_id = clean_text(anchor_el.get("id", "")) if anchor_el else ""
            message_id = anchor_id if anchor_id else native_post_id

            author, user_id, native_user_id = extract_author_and_user_id(block)
            date_display = extract_visible_date_from_block(block)
            date_iso = extract_iso_date_from_block(block)

            body = extract_body_from_block(block)

            # HARD FILTER FOR THE EXACT BUG YOU SHOWED
            if is_bad_body(body):
                continue

            # Another protection:
            # if body is only the toolbar copied text, skip it
            if clean_text(body) == "Alerter Copier le lien Le lien a été copié dans votre presse-papier":
                continue

            all_posts.append({
                "author": author,
                "user_id": user_id,
                "native_user_id": native_user_id,
                "date": date_display,
                "date_iso": date_iso,
                "body": body,
                "likes_count": 0,
                "dislikes_count": 0,
                "thread_id": thread_id,
                "thread_page_number": page_num,
                "post_sequence_on_page": len(all_posts) + 1,
                "message_id": message_id,
                "native_post_id": native_post_id,
                "anchor_id": anchor_id,
            })

    return {
        "thread_id": thread_id,
        "thread_title_detail": thread_title_detail,
        "thread_pages_count": pages_count,
        "posts": all_posts,
    }

# MAIN SCRAPER: DOCTISSIMO
# =========================================================

def scrape_doctissimo_forum(cfg: dict, session: requests.Session, posts_file: Path, errors_file: Path):
    global should_stop

    source_id = cfg["source_id"]
    timeout = cfg.get("request", {}).get("timeout_seconds", 30)
    sleep_seconds = cfg.get("request", {}).get("sleep_seconds", 1.0)
    max_listing_pages = cfg.get("pagination", {}).get("max_listing_pages", 1)

    seen_thread_ids = set()
    total_threads_processed = 0
    total_posts_extracted = 0

    print(f"🚀 Starting scraper for {source_id}")
    print(f"📊 Max listing pages: {max_listing_pages}")
    print(f"⏱️ Sleep: {sleep_seconds}s")
    print("=" * 60)

    for seed_url in cfg.get("start_urls", []):
        print(f"\n🌐 Seed URL: {seed_url}")

        for page_num in range(1, max_listing_pages + 1):
            if should_stop:
                break

            listing_url = build_doctissimo_listing_url(seed_url, page_num)

            try:
                print(f"\n📄 Page {page_num}/{max_listing_pages}: {listing_url}")
                listing_html = fetch_html(session, listing_url, timeout)
                listing_soup = BeautifulSoup(listing_html, "lxml")

                threads = extract_threads_from_listing(listing_soup, listing_url)

                if not threads:
                    print(f"   ⚠️ No threads found on page {page_num}")
                    break

                print(f"   ✅ Found {len(threads)} potential threads on this page")

                for i, thread_meta in enumerate(threads, start=1):
                    if should_stop:
                        break

                    thread_id = thread_meta["thread_id"]
                    thread_url = thread_meta["thread_url"]

                    if thread_id in seen_thread_ids:
                        print(f"   ⏭️ Thread {i}/{len(threads)} already processed: {thread_id}")
                        continue
                    seen_thread_ids.add(thread_id)

                    try:
                        print(f"\n   🔗 Thread {i}/{len(threads)}: {thread_meta['thread_title'][:70]}")

                        parsed = parse_doctissimo_thread(
                            session=session,
                            thread_url=thread_url,
                            timeout=timeout,
                            sleep_seconds=sleep_seconds,
                        )

                        all_posts = parsed["posts"]
                        thread_title_detail = parsed["thread_title_detail"]
                        thread_pages_count = parsed["thread_pages_count"]

                        if not all_posts:
                            print("      ⚠️ No real posts found")
                            continue

                        opening_post = all_posts[0]
                        last_post = all_posts[-1]

                        opening_post_native_id = opening_post["native_post_id"]

                        for idx, post in enumerate(all_posts, start=1):
                            post["type"] = "post" if idx == 1 else "comment"
                            post["is_original_post"] = (idx == 1)

                            # director rule:
                            # opening post keeps its own native post id
                            # comments keep their own native post id in comment_id
                            # post_id for comments points to opening post native id
                            post["post_id"] = opening_post_native_id
                            post["comment_id"] = "" if idx == 1 else post["native_post_id"]

                        item = {
                            "source_id": source_id,
                            "source_mode": "forum_doctissimo",
                            "thread_id": parsed["thread_id"],
                            "thread_url_id": parsed["thread_id"],
                            "thread_title": thread_meta["thread_title"],
                            "thread_title_detail": thread_title_detail or thread_meta["thread_title"],
                            "thread_url": thread_url,
                            "last_page": thread_pages_count,
                            "thread_starter": opening_post["author"],
                            "thread_starter_id": opening_post["user_id"],
                            "opening_post_id": opening_post["native_post_id"],
                            "opening_message_id": opening_post["message_id"],
                            "opening_post_date": opening_post["date"],
                            "opening_post_body": opening_post["body"],
                            "listing_author": opening_post["author"],
                            "listing_author_id": opening_post["user_id"],
                            "replies_count": max(len(all_posts) - 1, 0),
                            "views_count": 0,
                            "last_message_date": last_post["date"],
                            "last_message_author": last_post["author"],
                            "last_message_author_id": last_post["user_id"],
                            "last_message_id": last_post["message_id"],
                            "thread_pages_count": thread_pages_count,
                            "posts_count": len(all_posts),
                            "comments_count": max(len(all_posts) - 1, 0),
                            "likes_total": sum(p.get("likes_count", 0) for p in all_posts),
                            "posts": all_posts,
                        }

                        append_jsonl(posts_file, item)

                        total_threads_processed += 1
                        total_posts_extracted += len(all_posts)

                        print(f"      ✅ {len(all_posts)} real posts extracted")
                        print(f"         👤 Starter: {opening_post['author']} | Last: {last_post['author']}")

                    except Exception as e:
                        print(f"      ❌ Thread error: {e}")
                        append_jsonl(errors_file, {
                            "source_id": source_id,
                            "thread_url": thread_url,
                            "error": str(e),
                            "timestamp": datetime.now().isoformat(),
                        })

                    time.sleep(sleep_seconds)

            except Exception as e:
                print(f"   ❌ Listing error on page {page_num}: {e}")
                append_jsonl(errors_file, {
                    "source_id": source_id,
                    "listing_url": listing_url,
                    "error": str(e),
                    "timestamp": datetime.now().isoformat(),
                })

            time.sleep(sleep_seconds)

        if should_stop:
            break

    print("\n" + "=" * 60)
    print("🎉 COMPLETE!")
    print(f"📊 Total threads: {total_threads_processed}")
    print(f"📊 Total posts: {total_posts_extracted}")
    print(f"📁 Output: {posts_file}")


# =========================================================
# ENTRY POINT
# =========================================================

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--config", required=True, help="Path to JSON config")
    args = parser.parse_args()

    cfg = load_config(args.config)
    posts_file, errors_file = build_output_files(cfg)

    session = requests.Session()
    session.headers.update(cfg.get("request", {}).get("headers", {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36"
    }))

    retry = Retry(
        total=3,
        connect=3,
        read=3,
        backoff_factor=1,
        status_forcelist=[429, 500, 502, 503, 504],
        allowed_methods=["HEAD", "GET", "OPTIONS"]
    )
    adapter = HTTPAdapter(max_retries=retry)
    session.mount("http://", adapter)
    session.mount("https://", adapter)

    mode = cfg.get("mode", "")

    if mode == "forum_doctissimo":
        scrape_doctissimo_forum(cfg, session, posts_file, errors_file)
    else:
        print(f"Mode not supported by this file: {mode}")

    print(f"\n📁 Output: {posts_file}")
    print(f"📁 Errors: {errors_file}")


if __name__ == "__main__":
    main()
